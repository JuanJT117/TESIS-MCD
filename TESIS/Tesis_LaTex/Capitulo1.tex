\begin{definicion}
	Un número entero es primo si y sólo si tiene exactamente cuatro divisores distintos.
\end{definicion}



\begin{teorema}[Fermat]
	No existen $x,y,z \in \mathbb{Z}$ (que no sean los triviales) tales que $x^n + y^n = z^n$ para $n \geq 3$.
\end{teorema}
